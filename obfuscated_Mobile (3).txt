
local function run_code(str)
    local result = ""
    for i = 1, #str do
        local byte = string.byte(str, i)
        local offset = ((i-1) % 10) + 1
        result = result .. string.char((byte - offset - 32) % 126 + 32)
    end
    return load(result)()
end
